\begin{frame}[containsverbatim]
\frametitle{Wie installiere \& entferne ich Pakete unter Ubuntu?}

  \begin{itemize}
   \item Update der Paketquellen: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get update}}}

   \vspace{1.0cm}

   \item Suche nach Paketen: \Ovalbox{\texttt{\textcolor{ubuntu}{(sudo) apt-cache search PAKETNAME}}}
   
   \vspace{1.0cm}

   \item Installation von Paketen: \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get install PAKETNAME1 PAKETNAME2}}}
  \end{itemize}

\end{frame}


%--------------------------------------------------------------------------------------%
%%%Folie 12



\begin{frame}[containsverbatim]
\frametitle{Wie installiere \& entferne ich Pakete unter Ubuntu?}

  \begin{itemize}
   \item Installierte Pakete auflisten: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{dpkg -l | grep PAKETNAME}}}

   \vspace{1.0cm}

   \item Entfernen, aber Einstellungen bleiben erhalten: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get remove PAKETNAME}}}
   
   \vspace{1.0cm}

   \item Vollständiges Entfernen samt Einstellungen: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get purge PAKETNAME}}}
  \end{itemize}

\end{frame}



%--------------------------------------------------------------------------------------%


%%%Folie 14



\begin{frame}[containsverbatim]
\frametitle{Wie installiere \& entferne ich Pakete unter Ubuntu?}

  \begin{itemize}
   \item Bereinigung des lokalen Caches von APT: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get clean}}} \\
   \textsl{\textcolor{dunkelgrau.65}{Aus dem lokalen Cache werden alle heruntergeladene Pakete gelöscht}}

%%%Es entfernt alles außer der Sperrdatei aus /var/cache/apt/archives/ und /var/cache/apt/archives/partial/


   \vspace{0.2cm}

   \item \textquotedblleft Intelligente\textquotedblright Bereinigung des lokalen Caches: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get autoclean}}} \\
   \textsl{\textcolor{dunkelgrau.65}{Es werden nur Dateien gelöscht, die nicht mehr heruntergeladen werden und nutzlos sind}}
   
   \vspace{0.2cm}

   \item Entfernen von nicht mehr benötigten Paketen: \\
   \Ovalbox{\texttt{\textcolor{ubuntu}{sudo apt-get autoremove}}} \\
   \textsl{\textcolor{dunkelgrau.65}{Es werden Pakete entfernt, die automatisch installiert wurden, um Abhängigkeiten zu erfüllen, jedoch nicht mehr gebraucht werden}}
  \end{itemize}

\end{frame}

\begin{frame}

\frametitle{Begriffserklärung}
	  
      \begin{itemize}
	\item \textbf{Repository:} \textcolor{dunkelgrau.60}{Softwarearchiv mit Paketen speziell für eine Linux-Distribution. Das Paketmanagement kennt nur Pakete aus Repositories die in der Datei \texttt{\textcolor{dark-violet}{/etc/apt/sources.list}} eingetragen sind oder als eigene Liste mit der Endung \textquotedblleft.list\textquotedblright im Verzeichnis \texttt{\textcolor{blue}{/etc/apt/sources.d}} !}

      \vspace{1.0cm}

	\item \textbf{Paketabhängigkeiten:} \textcolor{dunkelgrau.60}{Programme setzen jeweils bestimmte Pakete voraus. Die Paketverwaltung hat die Aufgabe diese Abhängigkeiten zu managen. Werden die Paketabhängigkeiten verletzt, weil zwei unterschiedliche Programme ein bestimmtes Paket in jeweils anderer Version benötigen, wird eines dieser Programme entweder entfernt, bzw. dieses erst gar nicht installiert.}
      \end{itemize}

\end{frame}
